function CheckFolder($dir)
{
    if (-not (Test-Path $dir)){
        echo "Directory does not exist!"
        
        try{
            New-Item -Path $dir -ItemType Directory -ErrorAction Stop | Out-Null
        }
        catch{
            Write-Error -Message "Unable to Create Directory '$dir'. Error was: $_" -ErrorAction Stop
        }
        echo "Successfuly Created Directory"
    }
    else {
        echo "Directory exists!"
    }
}


function DisplayFolderInfo([string]$dir){
    
    $files = dir $dir -Recurse | where {!$_.PSIsContainer}
    $totals = $files | Measure-Object -Property length -Sum
    $stats = "" | Select path, count, size
    $stats.path = $dir
    $stats.count = $totals.count
    $stats.size = [math]::Round($totals.sum/1MB,3)
    return $stats
}
